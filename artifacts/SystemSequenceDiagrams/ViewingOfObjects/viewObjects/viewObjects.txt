@startuml
autonumber

actor User
participant "<u>:System" as S
collections "<u>:ObjectCatalog" as OC
participant "<u>:ObjectOfInterest" as OI

User -> S: login(userId, password)
S -> S: validateCredentials(userId, password)
S --> User: confirm Login

User -> S: viewObjects()
S -> OC: getAvailableObjects()

loop for each object
    OC -> OI: retrieveObjectDetails(objectId)
    OI --> OC: object
end

OC --> S: returnObjectsList([object1, object2, ...])
S --> User: displayObjects([object1, object2, ...])

@enduml
