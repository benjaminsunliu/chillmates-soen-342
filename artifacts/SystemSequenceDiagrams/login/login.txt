autonumber

actor User
participant "<u>:Institution" as S
collections "<u>:UserCatalog" as UC
participant "<u>:User" as U

User -> S: login(email, password)
S -> UC: findByEmail(email)
UC --> S: User

S -> U: getPassword()
S -> S: comparePasswords(password, U.getPassword())

S --> User: Login successful, return User

@enduml
